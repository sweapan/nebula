249 template<class TYPE> void
250 QuadTree<TYPE>::Node::Setup(QuadTree* tree, uchar _level, ushort _col, ushort _row)
251 {
252  #if NEBULA_BOUNDSCHECKS
253  n_assert(tree);
254  n_assert(this->level == -1);
255  n_assert(_col < (1 << _level));
256  n_assert(_row < (1 << _level));
257  #endif
258 
259  // store address
260  this->level = _level;
261  this->col = _col;
262  this->row = _row;
263 
264  // update my bounding box
265  float levelFactor = float(1 << (tree->treeDepth - 1 - this->level));
266  static Math::vector center;
267  static Math::vector extent;
268  const Math::vector& baseSize = tree->baseNodeSize;
269  const Math::bbox& treeBox = tree->boundingBox;
270  Math::vector treeSize = treeBox.size();
271  Math::vector treeCenter = treeBox.center();
272 
273  center.set(treeCenter.x + (((this->col + 0.5f) * levelFactor * baseSize.x) - (treeSize.x * 0.5f)),
274  treeCenter.y,
275  treeCenter.z + (((this->row + 0.5f) * levelFactor * baseSize.z) - (treeSize.z * 0.5f)));
276 
277  extent.set(levelFactor * baseSize.x * 0.5f,
278  treeBox.extents().y,
279  levelFactor * baseSize.z * 0.5f );
280 
281  this->box.set(center, extent);
282 
283  // recurse into children
284  uchar childLevel = this->level + 1;
285  if (childLevel < tree->treeDepth)
286  {
287  ushort i;
288  for (i = 0; i < 4; i++)
289  {
290  ushort childCol = 2 * this->col + (i & 1);
291  ushort childRow = 2 * this->row + ((i & 2) >> 1);
292  IndexT childIndex = tree->GetNodeIndex(childLevel, childCol, childRow);
293  this->children[i] = &(tree->nodeArray[childIndex]);
294  this->children[i]->Setup(tree, childLevel, childCol, childRow);
295  }
296  }
297 }
