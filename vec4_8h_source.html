1041 {

1043 
1044  NEBULA_ALIGN16 unsigned int elem[4] = { i0, i1, i2, i3 };
1045  __m128i vControl = _mm_load_si128(reinterpret_cast<const __m128i*>(&elem[0]));
1046

1048  vControl = _mm_and_si128(vControl, three);
1049

1051  __m128 shuffled2 = _mm_permutevar_ps(v1.vec, vControl);
