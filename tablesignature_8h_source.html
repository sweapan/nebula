101 TableSignature::Setup(PropertyId const* propertyBuffer, SizeT num)
102 {
103  if (num > 0)
104  {
105  Util::Array<PropertyId> descriptors(propertyBuffer, num);
106  descriptors.Sort();
107  PropertyId const last = *(descriptors.End() - 1);
108  n_assert(last != PropertyId::Invalid());
109  uint const largestBit = last.id;
110  this->size = (largestBit / 128) + 1;
111  this->mask = (__m128i*)Memory::Alloc(Memory::HeapType::ObjectHeap, this->size * 16);
112 
113  for (int i = 0; i < this->size; i++)
114  this->mask[i] = _mm_setzero_si128();
115 
116  alignas(16) uint64_t partialMask[2] = { 0, 0 };
117  // offsets start at first values offset
118  uint64_t offset = descriptors[0].id / 128;
119  uint64_t prevOffset = offset;
120  for (PropertyId i : descriptors)
121  {
122  n_assert(i != PropertyId::Invalid());
123  offset = i.id / 128;
124  if (offset != prevOffset)
125  {
126  // Write value and reset partial mask
127  *(this->mask + prevOffset) = _mm_set_epi64x(partialMask[0], partialMask[1]);
128  prevOffset = offset;
129  memset(&partialMask, 0, sizeof(partialMask));
130  }

132  partialMask[mod / 64] |= 1ull << mod;
133  }

136  }
