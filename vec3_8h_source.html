521

818 {

820 
821  NEBULA_ALIGN16 unsigned int elem[4] = { i0, i1, i2, 7 };
822  __m128i vControl = _mm_load_si128(reinterpret_cast<const __m128i*>(&elem[0]));
823 
824  __m128i vSelect = _mm_cmpgt_epi32(vControl, three);
825  vControl = _mm_and_si128(vControl, three);
826 
827  __m128 shuffled1 = _mm_permutevar_ps(v0.vec, vControl);
828  __m128 shuffled2 = _mm_permutevar_ps(v1.vec, vControl);
829

831  __m128 masked2 = _mm_and_ps(_mm_castsi128_ps(vSelect), shuffled2);
832 
833  return _mm_or_ps(masked1, masked2);
