514  l = _mm_rsqrt_ps(l);
515  l = _mm_mul_ss(_mm_shuffle_ps(l, l, _MM_SHUFFLE(0, 0, 0, 0)), _mm_shuffle_ps(l, l, _MM_SHUFFLE(1, 1, 1, 1)));
516 
517 
518  __m128 dot = _mm_mul_ps(v0.vec, v1.vec);
519  dot = _mm_add_ps(_mm_shuffle_ps(dot, dot, _MM_SHUFFLE(0, 0, 0, 0)),

815 
816  NEBULA_ALIGN16 unsigned int elem[4] = { i0, i1, i2, 7 };
817  __m128i vControl = _mm_load_si128(reinterpret_cast<const __m128i*>(&elem[0]));
818

820  vControl = _mm_and_si128(vControl, three);

823  __m128 shuffled2 = _mm_permutevar_ps(v1.vec, vControl);
