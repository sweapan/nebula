297 inline TextureUsage
298 TextureUsageFromString(const Util::String& string)
299 {
300  Util::Array<Util::String> comps = string.Tokenize("|");
301  TextureUsage usage = TextureUsage(0x0);
302 
303  for (IndexT i = 0; i < comps.Size(); i++)
304  {
305  if (comps[i] == "Sample") usage |= SampleTexture;
306  else if (comps[i] == "Render") usage |= RenderTexture;
307  else if (comps[i] == "ReadWrite") usage |= ReadWriteTexture;
308  else if (comps[i] == "TransferSource") usage |= TransferTextureSource;
309  else if (comps[i] == "TransferDestination") usage |= TransferTextureDestination;
310  }
311 
312  return usage;
313 }
