101 template<class TYPE> bool
102 Extrapolator<TYPE>::AddSample(Timing::Time packetTime, Timing::Time curTime, const TYPE& pos, const TYPE& vel)
103 {
104  if (!Estimates(packetTime, curTime)) {
105  return false;
106  }
107  this->lastPacketPos = pos;
108  this->lastPacketTime = packetTime;
109  this->ReadValue(curTime, this->snapPos);
110  this->aimTime = curTime + this->updateTime;
111  float dt = (float)(this->aimTime - packetTime);
112  this->snapTime = curTime;
113  this->aimPos = pos + vel * dt;
114 
115  // I now have two positions and two times:
116  // this->aimPos / this->aimTime
117  // this->snapPos / this->snapTime
118  // I must generate the interpolation velocity based on these two samples.
119  // However, if this->aimTime is the same as this->snapTime, I'm in trouble. In that
120  // case, use the supplied velocity.
121  if (Math::abs(this->aimTime - this->snapTime) < N_TINY)
122  {
123  this->snapVel = vel;
124  }
125  else
126  {
127  float dt = (float)(1.0 / (this->aimTime - this->snapTime));
128  this->snapVel = (this->aimPos - this->snapPos) * dt;
129  }
130  return true;
131 }

222 template<class TYPE> bool
223 Extrapolator<TYPE>::Estimates(Timing::Time packet, Timing::Time cur)
224 {
225  if (packet <= this->lastPacketTime)
226  {
227  return false;
228  }
229 
230  // The theory is that, if latency increases, quickly
231  // compensate for it, but if latency decreases, be a
232  // little more resilient; this is intended to compensate
233  // for jittery delivery.
234  Timing::Time lat = cur - packet;
235  if (lat < 0) lat = 0;
236  if (lat > this->latency)
237  {
238  this->latency = (this->latency + lat) * 0.5;
239  }
240  else
241  {
242  this->latency = (this->latency * 7 + lat) * 0.125;
243  }
244 
245  // Do the same running average for update time.
246  // Again, the theory is that a lossy connection wants
247  // an average of a higher update time.
248  Timing::Time tick = packet - this->lastPacketTime;
249  if (tick > this->updateTime)
250  {
251  this->updateTime = (this->updateTime + tick) * 0.5;
252  }
253  else
254  {
255  this->updateTime = (this->updateTime * 7 + tick) * 0.125;
256  }
257 
258  return true;
259 }
