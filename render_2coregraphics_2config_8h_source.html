129 inline ShaderVisibility
130 ShaderVisibilityFromString(const Util::String& str)
131 {
132  Util::Array<Util::String> components = str.Tokenize("|");
133  CoreGraphics::ShaderVisibility ret = InvalidVisibility;
134  IndexT i;
135  for (i = 0; i < components.Size(); i++)
136  {
137  const Util::String& component = components[i];
138  if (component == "VS") ret |= VertexShaderVisibility;
139  else if (component == "HS") ret |= HullShaderVisibility;
140  else if (component == "DS") ret |= DomainShaderVisibility;
141  else if (component == "GS") ret |= GeometryShaderVisibility;
142  else if (component == "PS") ret |= PixelShaderVisibility;
143  else if (component == "CS") ret |= ComputeShaderVisibility;
144  }
145 
146  return ret;
147 }
