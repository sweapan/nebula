527 template<class TYPE> void
528 Array<TYPE>::Move(IndexT fromIndex, IndexT toIndex)
529 {
530  #if NEBULA_BOUNDSCHECKS
531  n_assert(this->elements);
532  n_assert(fromIndex < this->count);
533  #endif
534 
535  // nothing to move?
536  if (fromIndex == toIndex)
537  {
538  return;
539  }
540 
541  // compute number of elements to move
542  SizeT num = this->count - fromIndex;
543 
544  // check if array needs to grow
545  SizeT neededSize = toIndex + num;
546  while (neededSize > this->capacity)
547  {
548  this->Grow();
549  }
550 
551  if (fromIndex > toIndex)
552  {
553  // this is a backward move
554  this->MoveRange(&this->elements[toIndex], &this->elements[fromIndex], num);
555  this->DestroyRange(fromIndex + num - 1, this->count);
556  }
557  else
558  {
559  // this is a forward move
560  int i; // NOTE: this must remain signed for the following loop to work!!!
561  for (i = num - 1; i >= 0; --i)
562  {
563  this->elements[toIndex + i] = this->elements[fromIndex + i];
564  }
565 
566  // destroy freed elements
567  this->DestroyRange(fromIndex, toIndex);
568  }
569 
570  // adjust array size
571  this->count = toIndex + num;
