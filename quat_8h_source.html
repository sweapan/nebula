434 slerp(const quat& q1, const quat& q2, scalar t)
435 {
436  __m128 to;
437 
438  float qdot = dot(q1, q2);
439  // flip when negative angle
440  if (qdot < 0)
441  {
442  qdot = -qdot;
443  to = _mm_mul_ps(q2.vec, _mm_set_ps1(-1.0f));
444  }
445  else
446  {
447  to = q2.vec;
448  }
449 
450  // just lerp when angle between is narrow
451  if (qdot < 0.95f)
452  {
453  //dont break acos
454  float clamped = Math::clamp(qdot, -1.0f, 1.0f);
455  float angle = Math::acos(clamped);
456 
457  float sin_angle = Math::sin(angle);
458  float sin_angle_t = Math::sin(angle * t);
459  float sin_omega_t = Math::sin(angle * (1.0f - t));
460 
461  __m128 s0 = _mm_set_ps1(sin_omega_t);
462  __m128 s1 = _mm_set_ps1(sin_angle_t);
463  __m128 sin_div = _mm_set_ps1(1.0f / sin_angle);
464  return _mm_mul_ps(_mm_add_ps(_mm_mul_ps(q1.vec, s0), _mm_mul_ps(to, s1)), sin_div);
465  }
466  else
467  {
468 
469  float scale0 = 1.0f - t;
470  float scale1 = t;
471  __m128 s0 = _mm_set_ps1(scale0);
472  __m128 s1 = _mm_set_ps1(scale1);
473 
474  return _mm_add_ps(_mm_mul_ps(q1.vec, s0), _mm_mul_ps(to, s1));
475  }
