364 __forceinline scalar
365 determinant(const mat4& m)
366 {
367  __m128 Va,Vb,Vc;
368  __m128 r1,r2,r3,tt,tt2;
369  __m128 sum,Det;
370 
371  __m128 _L1 = m.r[0].vec;
372  __m128 _L2 = m.r[1].vec;
373  __m128 _L3 = m.r[2].vec;
374  __m128 _L4 = m.r[3].vec;
375  // Calculating the minterms for the first line.
376 
377  // _mm_ror_ps is just a macro using _mm_shuffle_ps().
378  tt = _L4; tt2 = mm_ror_ps(_L3,1);
379  Vc = _mm_mul_ps(tt2, mm_ror_ps(tt,0)); // V3' dot V4
380  Va = _mm_mul_ps(tt2, mm_ror_ps(tt,2)); // V3' dot V4"
381  Vb = _mm_mul_ps(tt2, mm_ror_ps(tt,3)); // V3' dot V4^
382 
383  r1 = _mm_sub_ps(mm_ror_ps(Va,1), mm_ror_ps(Vc,2)); // V3" dot V4^ - V3^ dot V4"
384  r2 = _mm_sub_ps(mm_ror_ps(Vb,2), mm_ror_ps(Vb,0)); // V3^ dot V4' - V3' dot V4^
385  r3 = _mm_sub_ps(mm_ror_ps(Va,0), mm_ror_ps(Vc,1)); // V3' dot V4" - V3" dot V4'
386 
387  tt = _L2;
388  Va = mm_ror_ps(tt,1); sum = _mm_mul_ps(Va,r1);
389  Vb = mm_ror_ps(tt,2); sum = _mm_add_ps(sum,_mm_mul_ps(Vb,r2));
390  Vc = mm_ror_ps(tt,3); sum = _mm_add_ps(sum,_mm_mul_ps(Vc,r3));
391 
392  // Calculating the determinant.
393  Det = _mm_mul_ps(sum,_L1);
394  Det = _mm_add_ps(Det,_mm_movehl_ps(Det,Det));
395 
396  // Calculating the minterms of the second line (using previous results).
397  tt = mm_ror_ps(_L1,1); sum = _mm_mul_ps(tt,r1);
398  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
399  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
400 
401  // Testing the determinant.
402  Det = _mm_sub_ss(Det,_mm_shuffle_ps(Det,Det,1));
403  return vec4(Det).x;
404 }

409 __forceinline mat4
410 inverse(const mat4& m)
411 {
412  __m128 Va,Vb,Vc;
413  __m128 r1,r2,r3,tt,tt2;
414  __m128 sum,Det,RDet;
415  __m128 trns0,trns1,trns2,trns3;
416 
417  const __m128i pnpn = _mm_setr_epi32(0x00000000, static_cast<int>(0x80000000), 0x00000000, static_cast<int>(0x80000000));
418  const __m128i npnp = _mm_setr_epi32(static_cast<int>(0x80000000), 0x00000000, static_cast<int>(0x80000000), 0x00000000);
419  const __m128 zeroone = _mm_setr_ps(1.0f, 0.0f, 0.0f, 1.0f);
420 
421  __m128 _L1 = m.r[0].vec;
422  __m128 _L2 = m.r[1].vec;
423  __m128 _L3 = m.r[2].vec;
424  __m128 _L4 = m.r[3].vec;
425  // Calculating the minterms for the first line.
426 
427  // _mm_ror_ps is just a macro using _mm_shuffle_ps().
428  tt = _L4; tt2 = mm_ror_ps(_L3,1);
429  Vc = _mm_mul_ps(tt2, mm_ror_ps(tt,0)); // V3'dot V4
430  Va = _mm_mul_ps(tt2, mm_ror_ps(tt,2)); // V3'dot V4"
431  Vb = _mm_mul_ps(tt2, mm_ror_ps(tt,3)); // V3' dot V4^
432 
433  r1 = _mm_sub_ps(mm_ror_ps(Va,1), mm_ror_ps(Vc,2)); // V3" dot V4^ - V3^ dot V4"
434  r2 = _mm_sub_ps(mm_ror_ps(Vb,2), mm_ror_ps(Vb,0)); // V3^ dot V4' - V3' dot V4^
435  r3 = _mm_sub_ps(mm_ror_ps(Va,0), mm_ror_ps(Vc,1)); // V3' dot V4" - V3" dot V4'
436 
437  tt = _L2;
438  Va = mm_ror_ps(tt,1); sum = _mm_mul_ps(Va,r1);
439  Vb = mm_ror_ps(tt,2); sum = _mm_add_ps(sum,_mm_mul_ps(Vb,r2));
440  Vc = mm_ror_ps(tt,3); sum = _mm_add_ps(sum,_mm_mul_ps(Vc,r3));
441 
442  // Calculating the determinant.
443  Det = _mm_mul_ps(sum,_L1);
444  Det = _mm_add_ps(Det,_mm_movehl_ps(Det,Det));
445 
446 
447  __m128 mtL1 = _mm_xor_ps(sum, _mm_castsi128_ps(pnpn));
448 
449  // Calculating the minterms of the second line (using previous results).
450  tt = mm_ror_ps(_L1,1); sum = _mm_mul_ps(tt,r1);
451  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
452  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
453  __m128 mtL2 = _mm_xor_ps(sum, _mm_castsi128_ps(npnp));
454 
455  // Testing the determinant.
456  Det = _mm_sub_ss(Det,_mm_shuffle_ps(Det,Det,1));
457 
458  // Calculating the minterms of the third line.
459  tt = mm_ror_ps(_L1,1);
460  Va = _mm_mul_ps(tt,Vb); // V1' dot V2"
461  Vb = _mm_mul_ps(tt,Vc); // V1' dot V2^
462  Vc = _mm_mul_ps(tt,_L2); // V1' dot V2
463 
464  r1 = _mm_sub_ps(mm_ror_ps(Va,1), mm_ror_ps(Vc,2)); // V1" dot V2^ - V1^ dot V2"
465  r2 = _mm_sub_ps(mm_ror_ps(Vb,2), mm_ror_ps(Vb,0)); // V1^ dot V2' - V1' dot V2^
466  r3 = _mm_sub_ps(mm_ror_ps(Va,0), mm_ror_ps(Vc,1)); // V1' dot V2" - V1" dot V2'
467 
468  tt = mm_ror_ps(_L4,1); sum = _mm_mul_ps(tt,r1);
469  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
470  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
471  __m128 mtL3 = _mm_xor_ps(sum, _mm_castsi128_ps(pnpn));
472 
473  // Dividing is FASTER than rcp_nr! (Because rcp_nr causes many register-memory RWs).
474  RDet = _mm_div_ss(zeroone, Det); // TODO: just 1.0f?
475  RDet = _mm_shuffle_ps(RDet,RDet,0x00);
476 
477  // Devide the first 12 minterms with the determinant.
478  mtL1 = _mm_mul_ps(mtL1, RDet);
479  mtL2 = _mm_mul_ps(mtL2, RDet);
480  mtL3 = _mm_mul_ps(mtL3, RDet);
481 
482  // Calculate the minterms of the forth line and devide by the determinant.
483  tt = mm_ror_ps(_L3,1); sum = _mm_mul_ps(tt,r1);
484  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r2));
485  tt = mm_ror_ps(tt,1); sum = _mm_add_ps(sum,_mm_mul_ps(tt,r3));
486  __m128 mtL4 = _mm_xor_ps(sum, _mm_castsi128_ps(npnp));
487  mtL4 = _mm_mul_ps(mtL4, RDet);
488 
489  // Now we just have to transpose the minterms matrix.
490  trns0 = _mm_unpacklo_ps(mtL1,mtL2);
491  trns1 = _mm_unpacklo_ps(mtL3,mtL4);
492  trns2 = _mm_unpackhi_ps(mtL1,mtL2);
493  trns3 = _mm_unpackhi_ps(mtL3,mtL4);
494  _L1 = _mm_movelh_ps(trns0,trns1);
495  _L2 = _mm_movehl_ps(trns1,trns0);
496  _L3 = _mm_movelh_ps(trns2,trns3);
497  _L4 = _mm_movehl_ps(trns3,trns2);
498 
499  return mat4(vec4(_L1), vec4(_L2), vec4(_L3), vec4(_L4));
500 }
