132 RingAllocator<SYNCPOINTS>::Allocate(const SizeT size, RingAlloc& alloc)
133 {
134  n_assert(size <= this->size);
135  uint begin = this->currentInterval.upper;
136  uint end = begin + size;
137  if (end > this->size && this->freeInterval.lower >= this->freeInterval.upper)
138  {
139  // wrap around to beginning, which should be okay if the lower is higher than upper, meaning
140  // we popped a lower interval during Start() and can now reuse from the beginning of the buffer
141  alloc.offset = 0;
142  end = this->freeInterval.upper;
143  }
144  else if (end > this->freeInterval.upper)
145  {
146  // if our intervals are not wrapping, we can just check to see if we will go overboard
147  n_warning("Over-allocated RingAllocator!\n");
148  alloc.data = nullptr;

150  return false;
