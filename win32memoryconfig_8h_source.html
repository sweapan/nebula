122 {
123  // restore unaligned pointer
124  unsigned char* ptr = (unsigned char*) lpMem;
125  unsigned char* rawPtr = __HeapUnalignPointer16(ptr);
126 
127  // perform re-alloc, NOTE: if re-allocation can't happen in-place,
128  // we need to handle the allocation ourselves, in order not to destroy
129  // the original data because of different alignment!!!
130  ptr = (unsigned char*) ::HeapReAlloc(hHeap, (dwFlags | HEAP_REALLOC_IN_PLACE_ONLY), rawPtr, dwBytes + 16);
131  if (0 == ptr)
132  {
133  SIZE_T rawSize = ::HeapSize(hHeap, dwFlags, rawPtr);
134  // re-allocate manually because padding may be different!
135  ptr = (unsigned char*) ::HeapAlloc(hHeap, dwFlags, dwBytes + 16);
136  ptr = __HeapAlignPointerAndWritePadding16(ptr);
137  SIZE_T copySize = dwBytes <= (rawSize - 16) ? dwBytes : (rawSize - 16);
138  ::CopyMemory(ptr, lpMem, copySize);
